def execute(runner)
  start
  status = runner.execute(to_s) do |out, err|
    run_hook(:on_stdout, out) if out
    run_hook(:on_stderr, err) if err
  end
ensure
  finish(status) # nil if not called
end

def finish(status)
  @finished = Time.now
  @status   = status
  run_hook(:on_finish, @finished, @status)
  @status
end
